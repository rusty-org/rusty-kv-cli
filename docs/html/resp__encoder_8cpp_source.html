56std::vector<std::string> tokenize(const std::string& input) {
57 std::vector<std::string> tokens;
58 std::istringstream iss(input);
59 std::string token;
60
61 bool in_quotes = false;
62 std::string quoted_token;
63
64 while (iss >> token) {
65 // Check if token starts with a quote
66 if (!in_quotes && token.front() == '"') {
67 in_quotes = true;
68 quoted_token = token.substr(1); // Remove leading quote
69
70 // If it also ends with a quote, it's a complete quoted token
71 if (token.length() > 1 && token.back() == '"') {
72 in_quotes = false;
73 quoted_token = quoted_token.substr(0, quoted_token.length() - 1); // Remove trailing quote
74 tokens.push_back(quoted_token);
75 }
76 }
77 // Check if token ends with a quote
78 else if (in_quotes && token.back() == '"') {
79 in_quotes = false;
80 quoted_token += " " + token.substr(0, token.length() - 1); // Remove trailing quote
81 tokens.push_back(quoted_token);
82 }
83 // If we're in the middle of a quoted string
84 else if (in_quotes) {
85 quoted_token += " " + token;
86 }
87 // Regular token, not part of a quoted string
88 else {
89 tokens.push_back(token);
90 }
91 }
92
93 // If we're still in quotes after processing all tokens,
94 // add what we have (handles unclosed quotes)
95 if (in_quotes) {
96 tokens.push_back(quoted_token);
97 }
98
99 return tokens;
100}
