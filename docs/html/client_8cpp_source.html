33bool KvClient::connect(const std::string& host, int port) {
34 // @INFO Create socket
35 socket_fd = socket(AF_INET, SOCK_STREAM, 0);
36 if (socket_fd < 0) {
37 std::string error_msg = "Socket creation failed: " + std::string(strerror(errno));
38 Logger::error(error_msg);
39 return false;
40 }
41
42 // Set connection info
43 connectionInfo.host = host;
44 connectionInfo.port = port;
45
46 // @INFO Set up server address
47 struct sockaddr_in server_addr;
48 server_addr.sin_family = AF_INET;
49 server_addr.sin_port = htons(port);
50
51 if (inet_pton(AF_INET, host.c_str(), &server_addr.sin_addr) <= 0) {
52 std::string error_msg = "Invalid address: " + std::string(strerror(ECONNREFUSED));
53 Logger::error(error_msg);
54 close(socket_fd);
55 return false;
56 }
57
58 // @INFO Connect to server
59 if (::connect(socket_fd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
60 std::string error_msg = "Connection failed: " + std::string(strerror(errno));
61 Logger::error(error_msg);
62 close(socket_fd);
63 return false;
64 }
65
66 connected = true;
67 Logger::info("Connected to server at " + host + ":" + std::to_string(port));
68 this->addr = host + ":" + std::to_string(port);
69 return true;
70}
